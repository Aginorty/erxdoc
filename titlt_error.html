# Total error

def projection_error(camera_position_in_frame, camera_height, beta_, angle_from_optical_axis):
    point_ground_position_=camera_height*np.tan(angle_from_optical_axis)
    return(camera_position_in_frame+point_ground_position_)* \
        ((np.cos(beta_)+np.sin(beta_)*np.tan(beta_+angle_from_optical_axis))-1)
                                                        
plt.plot(np.rad2deg(angle_range),projection_error(L_m/2+0.5, H_m, beta, angle_range))
title = 'Projection error due tilt form obstacle under one wheel \n for a fixed obstacle height of 5cm \n on the right side of the machine'
plt.title(title)
plt.xlabel('angle from optical axis (deg)')
plt.ylabel('Projection error (m)')
plt.grid(True)
